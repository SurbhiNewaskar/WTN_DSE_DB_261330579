HANDS-ON ASSIGNMENT
Topic: Group Functions

 1: Find highest, lowest, sum and average salary of all employees Label columns Maximum, Minimum, Sum, Average

SELECT MAX(SAL) AS Maximum,
       MIN(SAL) AS Minimum,
       SUM(SAL) AS Sum,
       AVG(SAL) AS Average
FROM EMP;


 2: Same as above but round results to nearest whole number

SELECT ROUND(MAX(SAL)) AS Maximum,
       ROUND(MIN(SAL)) AS Minimum,
       ROUND(SUM(SAL)) AS Sum,
       ROUND(AVG(SAL)) AS Average
FROM EMP;


 3: Display minimum, maximum, sum and average salary for each job type

SELECT JOB,
       ROUND(MAX(SAL)) AS Maximum,
       ROUND(MIN(SAL)) AS Minimum,
       ROUND(SUM(SAL)) AS Sum,
       ROUND(AVG(SAL)) AS Average
FROM EMP
GROUP BY JOB;


 4: Display number of people with the same job

SELECT JOB, COUNT(*) AS "Number of People"
FROM EMP
GROUP BY JOB;


 5: Determine number of managers without listing them Label column Number of Managers

SELECT COUNT(DISTINCT MGR) AS "Number of Managers"
FROM EMP;


 6: Find difference between highest and lowest salary Label column DIFFERENCE

SELECT MAX(SAL) - MIN(SAL) AS DIFFERENCE
FROM EMP;


 7: Display manager number and salary of lowest paid employee for that manager Exclude unknown managers, exclude groups where min salary is 2000 or less Sort descending by salary

SELECT MGR, MIN(SAL) AS "Lowest Salary"
FROM EMP
WHERE MGR IS NOT NULL
GROUP BY MGR
HAVING MIN(SAL) > 2000
ORDER BY MIN(SAL) DESC;


8: Display total number of employees and number hired in 1980, 1981, 1982 With appropriate column headings

SELECT COUNT(*) AS "Total",
       SUM(CASE WHEN TO_CHAR(HIREDATE, 'YYYY') = '1980' THEN 1 ELSE 0 END) AS "1980",
       SUM(CASE WHEN TO_CHAR(HIREDATE, 'YYYY') = '1981' THEN 1 ELSE 0 END) AS "1981",
       SUM(CASE WHEN TO_CHAR(HIREDATE, 'YYYY') = '1982' THEN 1 ELSE 0 END) AS "1982"
FROM EMP;
